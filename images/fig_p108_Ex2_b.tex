\begin{tikzpicture}[angle/.style={black,thick},]
\tikzmath{\r=2.5;\h=5;\q=atan((\r/\h)); \kei=sin(\q)*sin(\q);\skei=sqrt(\kei);\pr=0.549306/1 ; \pp=1.61557/2.5;};
\q, \h,\pr, \pp,\kei,\skei;

    \pgfplotsset{ticks=none};
    \pgfplotsset{compat=1.12};
	\begin{axis}
		[
		axis equal=true,
		]
		%draw the cone
		\addplot3 [surf,	dashed,thick,	mesh/interior colormap=
		{bw}{color=(gray!10) color=(gray!50)},
		colormap= {bw}{color=(black!10) color=(white!10)},
		domain=0:360,
		y domain=0:\h,
		samples=40,
		samples y=10,
		z buffer=sort,]
		({\y*tan(\q)*sin(x)},{\y*tan(\q)*cos(x)},{y});
	
		%add a base circle to emphasize the cone shape (front and rear)
		\addplot3 [black, thick,dotted,samples=40,	samples y=0,	domain=-60:180-60]({\h*tan(\q)*sin(x)},{\h*tan(\q)*cos(x)},{\h});
		
		\addplot3 [black,dotted,samples=40,	samples y=0,	domain=180-60:360-60,]({\h*tan(\q)*sin(x)},{\h*tan(\q)*cos(x)},{\h});

		% add the visible "vector"
		\addplot3 [thick,decoration={markings, mark = at position 1 with {\arrow[scale = 1.5]{latex}}},postaction ={},samples=40,	samples y=0,	domain=0.0:0.4*2*3.141596]
		({\pp*cosh(\skei*x+\pr)*cos(deg(x))*sqrt(\kei)},
		{{\pp*cosh(\skei*x+\pr)*sin(deg(x))*sqrt(\kei)}},
		{\pp*cosh(\skei*x+\pr)*cos(\q)});
		
		% add the hidden "vector"
		\addplot3 [thick,dashed,decoration={markings, mark = at position 1 with {\arrow[scale = 1.5]{latex}}},postaction ={decorate},samples=40,	samples y=0,	domain=0.4*2*3.141596:1.7*3.141596]
		({\pp*cosh(\skei*x+\pr)*cos(deg(x))*sqrt(\kei)},
		{{\pp*cosh(\skei*x+\pr)*sin(deg(x))*sqrt(\kei)}},
		{\pp*cosh(\skei*x+\pr)*cos(\q)});
	
\end{axis}
\end{tikzpicture}\\
